==============================================================
  CLASHCODE - DOCKER COMMANDS REFERENCE
==============================================================

--------------------------------------------------------------
  STARTING & STOPPING
--------------------------------------------------------------

# Build and start everything (first time or after changes)
docker compose up --build -d

# Start without rebuilding
docker compose up -d

# Stop all containers (keeps data)
docker compose stop

# Stop and remove containers (keeps volumes/data)
docker compose down

# Stop, remove containers AND delete all data (DESTRUCTIVE!)
docker compose down -v


--------------------------------------------------------------
  MONITORING
--------------------------------------------------------------

# See all container statuses
docker compose ps -a

# View logs (all services)
docker compose logs

# View logs (specific service, last 50 lines, follow)
docker compose logs -f --tail=50 core
docker compose logs -f --tail=50 chat
docker compose logs -f --tail=50 ai
docker compose logs -f --tail=50 celery_worker
docker compose logs -f --tail=50 celery_beat
docker compose logs -f --tail=50 nginx
docker compose logs -f --tail=50 db
docker compose logs -f --tail=50 redis


--------------------------------------------------------------
  RESTART & REBUILD
--------------------------------------------------------------

# Restart a specific service
docker compose restart core
docker compose restart celery_beat
docker compose restart celery_worker
docker compose restart chat
docker compose restart ai
docker compose restart nginx

# Rebuild and restart a single service
docker compose up --build -d core
docker compose up --build -d ai
docker compose up --build -d chat


--------------------------------------------------------------
  DJANGO MANAGEMENT (CORE SERVICE)
--------------------------------------------------------------

# Run migrations
docker compose exec core python manage.py migrate

# Create superuser
docker compose exec core python manage.py createsuperuser

# Collect static files
docker compose exec core python manage.py collectstatic --noinput

# Django shell
docker compose exec core python manage.py shell

# Make new migrations after model changes
docker compose exec core python manage.py makemigrations

# Run tests
docker compose exec core python manage.py test


--------------------------------------------------------------
  SHELL ACCESS
--------------------------------------------------------------

# Open bash in a container
docker compose exec core bash
docker compose exec chat bash
docker compose exec ai bash

# Access PostgreSQL (core DB)
docker compose exec db psql -U postgres -d code_of_clan_db

# Access PostgreSQL (chat DB)
docker compose exec chat_db psql -U postgres -d chat_db

# Access Redis CLI
docker compose exec redis redis-cli


--------------------------------------------------------------
  DEBUGGING
--------------------------------------------------------------

# Check Docker Compose config for errors
docker compose config

# Inspect a specific container
docker inspect coc-core-1

# Check resource usage (CPU, memory, etc.)
docker stats

# View container's environment variables
docker compose exec core env
docker compose exec ai env
docker compose exec chat env


--------------------------------------------------------------
  CLEANUP
--------------------------------------------------------------

# Remove unused images
docker image prune -f

# Remove all stopped containers
docker container prune -f

# Remove unused volumes (DESTRUCTIVE - deletes data!)
docker volume prune -f

# Nuclear cleanup - everything unused (DESTRUCTIVE!)
docker system prune -af


--------------------------------------------------------------
  ACCESS POINTS
--------------------------------------------------------------

  Frontend:        http://localhost:5173  (or 5174 if 5173 in use)

  DynamoDB Admin:  http://localhost:8009
  DynamoDB Local:  http://localhost:8008


--------------------------------------------------------------
  SERVICES OVERVIEW
--------------------------------------------------------------

  nginx           - API Gateway (port 80)
  core            - Django REST API (port 8000 internal)
  celery_worker   - Background task worker
  celery_beat     - Scheduled task scheduler
  chat            - FastAPI WebSocket service (port 8001 internal)
  ai              - FastAPI AI service (port 8002 internal)
  db              - PostgreSQL (core database)
  chat_db         - PostgreSQL (chat database)
  redis           - Cache & message broker
  dynamodb        - DynamoDB Local (port 8008)
  dynamodb-admin  - DynamoDB Admin UI (port 8009)
  chroma          - ChromaDB vector store (port 8000 internal)
